def show
  id = params[:id]
  if quickstart = quickstarts.find{ |obj| obj['quickstart']['id'] == id }
    render_success(:ok, "quickstarts", [quickstart],  "Showing quickstart for '#{id}'")
  else
    render_error(:not_found, "Quickstart '#{id}' not found", 118)
  end
end
