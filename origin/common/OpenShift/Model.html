def to_xml(options = {})
  to_xml_opts = {:skip_types => true}
  to_xml_opts.merge!(options.slice(:builder, :skip_instruct))
  to_xml_opts[:root] = options[:tag_name] || self.class.name.underscore.gsub("_","-")
  self.attributes.to_xml(to_xml_opts)
end
