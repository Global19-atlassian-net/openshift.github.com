def self.with_no_cpu_limits(uuid)
  param = "cpu.cfs_quota_us"
  attrs = Attrs.new(uuid)
  full_cpu = attrs["cpu.cfs_period_us"]
  oldlimit = attrs[param]
  begin
    attrs[param]=full_cpu
    yield
  ensure
    attrs[param]=oldlimit
  end
end
