def action_log_tag_action
  case request.method
  when "GET"    then params[:id] ? "SHOW" : "LIST"
  when "POST"   then "ADD" 
  when "PUT"    then "UPDATE"
  when "DELETE" then "DELETE"
  else               "UNKNOWN"
  end
end
