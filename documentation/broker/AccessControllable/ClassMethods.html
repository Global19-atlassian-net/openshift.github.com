def members_of(acl, &block)
  self.in(
    _id:
      if Membership === acl
        acl.members.inject([]) do |a, m|
          next a unless member_type == m._type
          (next a unless yield m) if block_given?
          a << m._id
          a
        end
      else
        acl
      end
  )
end
