def deep_clear_default!(obj = self)
  if obj.is_a? Hash
    obj.default = nil if obj.default_proc
    obj.each {|k,v| deep_clear_default!(v)}
  elsif obj.is_a? Array
    obj.each {|v| deep_clear_default!(v)}
  end
  obj
end
