def callout
  childpidfile = "/var/run/openshift-node-web-proxy.pid"
  begin
    cpid = File.read(childpidfile).to_i
    if cpid != 0
      Process.kill("HUP", cpid)
    end
  rescue Errno::ENOENT
    # No child PID file
  rescue Errno::ESRCH
    # No such process
  end
end
