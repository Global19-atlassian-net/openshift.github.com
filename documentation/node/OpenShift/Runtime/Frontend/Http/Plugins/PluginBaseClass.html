def unprivileged_unidle
  begin
    http = Net::HTTP.new(SERVER_CONNECT_ADDR, SERVER_HTTP_PORT)
    http.open_timeout = 5
    http.read_timeout = 60
    http.use_ssl = false
    http.start do |client|
      resp = client.request_head('/', { 'Host' => @fqdn })
      resp.code
    end
  rescue
  end
end
